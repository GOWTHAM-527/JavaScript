/**
 * @param {number} x
 * @return {number}
 */
var reverse = function(x) {
    const isNegative = x < 0;
    let res = 0;
    let num = Math.abs(x);
    while (num > 0) {
        res = (res * 10) + (num % 10);
        num = Math.floor(num / 10);
    }
    if (isNegative) res *= -1;
    if (res < Math.pow(-2, 31) || res > Math.pow(2, 31) - 1) {
        return 0;
    }
    return res;
};
